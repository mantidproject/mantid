# Add the required dependencies

            PythonGeometryModule

            Kernel
            HistogramData
            Geometry
            ${PYTHON_LIBRARIES}
            ${POCO_LIBRARIES}
            ${Boost_LIBRARIES} )

elseif ( ${CMAKE_SYSTEM_NAME} STREQUAL "Linux" )
  set_target_properties( PythonAPIModule PROPERTIES
