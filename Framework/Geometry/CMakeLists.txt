                      PROPERTIES OUTPUT_NAME
                                 MantidGeometry
                                 COMPILE_DEFINITIONS
                                 IN_MANTID_GEOMETRY)

  set_target_properties(Geometry
                        PROPERTIES INSTALL_RPATH "@loader_path/../MacOS;@loader_path/../Frameworks")

  set_target_properties(Geometry
                        PROPERTIES INSTALL_RPATH "\$ORIGIN/../${LIB_DIR}")
endif()

  target_include_directories(Geometry SYSTEM PRIVATE ${OPENCASCADE_INCLUDE_DIR})
endif()

                      LINK_PRIVATE
                      ${MANTIDLIBS}
                      ${OPENGL_gl_LIBRARY}
                      ${OPENGL_glu_LIBRARY}
                      ${GSL_LIBRARIES}
                      ${NEXUS_LIBRARIES}
                      ${JSONCPP_LIBRARIES})

                      LINK_PUBLIC
                      ${TBB_LIBRARIES}
                      ${TBB_MALLOC_LIBRARIES})

    LINK_PRIVATE
    ${OPENGL_gl_LIBRARY}
    ${OPENGL_glu_LIBRARY})
